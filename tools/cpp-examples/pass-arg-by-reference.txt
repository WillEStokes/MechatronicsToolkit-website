/*In this example, PassByValue takes an integer x as an argument, which is passed by value.
This means that a copy of x is made and changes made to the copy are not reflected in the
original variable. On the other hand, PassByReference takes an integer x as an argument,
which is passed by reference. This means that a reference to x is created, allowing changes
made to the reference to be reflected in the original variable.*/

#include <iostream>

void PassByValue(int x) {
  x = 10;
}

void PassByReference(int &x) {
  x = 10;
}

int main() {
  int a = 5;
  PassByValue(a);
  std::cout << "After PassByValue, a = " << a << std::endl;

  int b = 5;
  PassByReference(b);
  std::cout << "After PassByReference, b = " << b << std::endl;

  return 0;
}
