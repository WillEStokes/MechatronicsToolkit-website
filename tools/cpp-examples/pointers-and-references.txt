/*This code declares an integer variable x and initializes it to 10. A pointer ptr to x and a
reference ref to x are declared and their values are printed. The value of x is then modified
through the pointer and reference, and the updated value is printed.*/

#include <iostream>
using namespace std;

int main() {
    int x = 10;
    int *ptr = &x; // pointer to x
    int &ref = x; // reference to x

    cout << "x = " << x << endl;
    cout << "ptr = " << ptr << ", *ptr = " << *ptr << endl;
    cout << "ref = " << ref << endl;

    *ptr = 20; // modify x through ptr
    cout << "x = " << x << endl;

    ref = 30; // modify x through ref
    cout << "x = " << x << endl;

    return 0;
}
