#This code calculates the inductive reactance, total impedance, cutoff frequency, phase shift,
#output voltage and gain of an RL circuit with a resistance of 1000 Ohms, an inductance of 100
#millihenries, a source voltage magnitude of 10 volts, and a frequency of 1000 Hz. You can change
#the input variables to simulate different scenarios.

import math

# Input variables
R = 1000  # Resistance (Ohms)
L = 0.1  # Inductance (Henry)
V = 10    # Source voltage magnitude (Volts)
f = 1000  # Frequency (Hz)

# Calculate inductive reactance
Xl = 2 * math.pi * f * L

# Calculate total impedance
Z = math.sqrt(R ** 2 + Xl ** 2)

# Calculate cutoff frequency
fc = R / (2 * math.pi * L)

# Calculate phase shift
phi = math.atan(Xl / R)

# Calculate output voltage and gain
Vout = V * Xl / Z
gain = 20 * math.log10(Vout / V)

# Output results
print("Inductive reactance = {:.2f} Ohms".format(Xl))
print("Total impedance = {:.2f} Ohms".format(Z))
print("Cutoff frequency = {:.2f} Hz".format(fc))
print("Phase shift = {:.2f} degrees".format(math.degrees(phi)))
print("Output voltage = {:.2f} Volts".format(Vout))
print("Gain = {:.2f} dB".format(gain))
